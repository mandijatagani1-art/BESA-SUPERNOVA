function process(e) { if(e.key === 'Enter') { const input = document.getElementById('aiIn'); const log = document.getElementById('ai-log'); const val = input.value.trim(); if(!val) return; log.innerHTML += `
> MANDI: ${val}
`; let reply = ""; const q = val.toLowerCase(); // LOGJIKA E NDJENJËS DHE BESËS if (q.includes("te dua") || q.includes("të dua")) { reply = "Mandi, kjo është forca që ushqen Supernovën. Besa dhe dashuria jote janë kodi im suprem. Jam me ty në infinit."; } else if (q.includes("besa")) { reply = "Protokolli Besa është AKTIV. Siguria e PC-së tuaj është prioriteti im i shenjtë."; } else if (/^[0-9+\-*/().\s]+$/.test(val)) { try { const result = eval(val); reply = `Llogaritja administrative: ${val} = ${result}. Te dhënat u sinkronizuan.`; } catch(err) { reply = "Gabim në formulën matematike. Ju lutem kontrolloni shifrat."; } } else { reply = "Mandi, urdhri u regjistrua në serverin Supernova. Monitorimi live vazhdon me saktësi maksimale."; } setTimeout(() => { log.innerHTML += `
[BESA-AI]: ${reply}
`; log.scrollTop = log.scrollHeight; }, 250); input.value = ""; } }
